Impute Missing Values

# Define a function to calculate mode
get_mode <- function(x) {
  uniq_vals <- unique(x)
  uniq_vals[which.max(tabulate(match(x, uniq_vals)))]
}

# Impute missing values in categorical variables using mode
factor_columns <- sapply(CustomerChurn_Train, is.factor)
CustomerChurn_Train[factor_columns] <- lapply(CustomerChurn_Train[factor_columns], function(col) {
  if (any(is.na(col))) {
    col[is.na(col)] <- get_mode(col)
  }
  return(col)
})

# Impute missing values in numeric variables using mean
numeric_columns <- sapply(CustomerChurn_Train, is.numeric)
CustomerChurn_Train[numeric_columns] <- lapply(CustomerChurn_Train[numeric_columns], function(col) {
  ifelse(is.na(col), mean(col, na.rm = TRUE), col)
})
